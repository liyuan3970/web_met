# 台州市气象快报统计系统
2022.4.4编辑数据

# 需要的三方包
pip install -r requirements.txt
碰到安装错误的包，可以使用conda安装

# 本地项目启动需要输入
redis-server
service mysql start

# 初始化数据库
python manage.py makemigrations zdz
python manage.py migrate

# 项目启动指令
python manage.py runapscheduler （定时任务）
python manage.py runserver 0.0.0.0:9001 （服务器）
